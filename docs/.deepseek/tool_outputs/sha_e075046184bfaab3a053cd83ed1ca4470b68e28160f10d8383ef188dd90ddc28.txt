   205│ def crf_loss(y_true, y_pred, transition_matrix):

   207│     y_true: 真实标签序列 [t1, t2, ..., tn]
   208│     y_pred: BiLSTM输出的发射分数矩阵 [n, num_tags]

   216│         real_score += y_pred[i, y_true[i]] + transition_matrix[y_true[i-1], y_true[i]]

   219│     # 使用前向算法
   220│     alpha = np.zeros(transition_matrix.shape[0])

   222│     alpha = y_pred[0]  # 初始状态为第一个字符的发射分数
   223│     # 递推

   225│         new_alpha = []
   226│         for tag in range(transition_matrix.shape[0]):
   227│             # 当前状态tag的得分 = 所有前一个状态prev_tag的得分 * 转移到当前状态的得分 * 当前发射得分
