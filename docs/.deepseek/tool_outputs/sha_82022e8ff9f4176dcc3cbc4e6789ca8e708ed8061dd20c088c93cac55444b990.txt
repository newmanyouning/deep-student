  1051│ ### 二、创建自动提交钩子
  1052│ 
  1053│ #### 方案1：最简单的 post-commit 钩子（手动触发后自动提交）

  1058│ # 2. 创建 post-commit 钩子

  1061│ echo "检测到新提交，自动保存所有修改..."

  1063│ git commit --amend -C HEAD --no-edit 2>/dev/null || true

  1070│ #### 方案2：创建文件监控钩子（推荐）
  1071│ 
  1072│ 创建一个独立的监控脚本：
  1073│ 
  1074│ ```bash
  1075│ # 1. 在项目根目录创建监控脚本
  1076│ cat > git-auto-commit.sh << 'EOF'
  1077│ #!/bin/bash
  1078│ # 自动提交监控脚本
  1079│ echo "🔍 开始监控文件变化..."
  1080│ 
  1081│ # 设置监控路径
  1082│ WATCH_PATH="."
  1083│ IGNORE_PATTERN='\.git/|node_modules/|\.log$|\.tmp$'
  1084│ 
  1085│ # 持续监控文件变化
  1086│ while true; do
  1087│     # 等待文件变化（macOS/Linux）
  1088│     if command -v fswatch &> /dev/null; then
  1089│         # macOS 使用 fswatch
  1090│         fswatch -1 -r -e "$IGNORE_PATTERN" "$WATCH_PATH" > /dev/null
  1091│     elif command -v inotifywait &> /dev/null; then
  1092│         # Linux 使用 inotifywait
  1093│         inotifywait -r -e modify,create,delete --exclude "$IGNORE_PATTERN" "$WATCH_PATH" > /dev/null
  1094│     else
  1095│         # 简单轮询（兼容所有系统）
  1096│         find "$WATCH_PATH" -type f -newer /tmp/last_check 2>/dev/null | head -1
  1097│         touch /tmp/last_check
  1098│         sleep 2
  1099│         continue
  1100│     fi
  1101│     
  1102│     # 等待一小段时间，避免多次触发
  1103│     sleep 1
  1104│     
  1105│     # 检查是否有未提交的修改
  1106│     if ! git diff --quiet || ! git diff --cached --quiet; then
  1107│         echo "📝 检测到文件变化，正在自动提交..."
  1108│         
  1109│         # 添加所有修改
  1110│         git add . 2>/dev/null
  1111│         
  1112│         # 生成时间戳
  1113│         TIMESTAMP=$(date "+%Y-%m-%d %H:%M:%S")
  1114│         
  1115│         # 提交
  1116│         git commit -m "🔄 自动提交: $TIMESTAMP" --quiet
  1117│         
  1118│         echo "✅ 已提交: $TIMESTAMP"
  1119│     fi
  1120│ done
  1121│ EOF
  1122│ 
  1123│ # 2. 给予执行权限
  1124│ chmod +x git-auto-commit.sh
  1125│ ```

  1207│ ### 四、更智能的自动提交方案
  1208│ 
  1209│ 创建一个更完善的监控系统：
  1210│ 
  1211│ ```bash
  1212│ # 创建智能监控脚本
  1213│ cat > git-auto-commit-pro.sh << 'EOF'

  1215│ # 智能自动提交系统

  1228│ # 清理锁文件
  1229│ trap "rm -f $LOCK_FILE; exit" SIGINT SIGTERM
  1230│ 
  1231│ log() {
  1232│     echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
  1233│ }
  1234│ 
  1235│ # 检查是否需要提交
  1236│ check_and_commit() {
  1237│     # 检查是否有未跟踪或修改的文件
  1238│     if [ -n "$(git status --porcelain)" ]; then
  1239│         log "检测到文件变化"
  1240│         
  1241│         # 智能判断提交类型
  1242│         CHANGES=$(git diff --name-only)
  1243│         
  1244│         if echo "$CHANGES" | grep -q "\.css$\|\.scss$\|\.less$"; then
  1245│             MSG="💄 样式更新"
  1246│         elif echo "$CHANGES" | grep -q "\.js$\|\.ts$\|\.jsx$\|\.tsx$"; then
  1247│             MSG="⚡️ 脚本更新"
  1248│         elif echo "$CHANGES" | grep -q "\.html$\|\.htm$"; then
  1249│             MSG="📄 页面更新"
  1250│         elif echo "$CHANGES" | grep -q "\.md$\|\.txt$"; then
  1251│             MSG="📝 文档更新"
  1252│         else
  1253│             MSG="🔧 自动提交"
  1254│         fi
  1255│         
  1256│         # 添加文件
  1257│         git add . 2>/dev/null
  1258│         
  1259│         # 提交
  1260│         if git commit -m "$MSG: $(date '+%Y-%m-%d %H:%M')" --quiet; then
  1261│             log "✅ 已提交: $MSG"
  1262│         fi
  1263│     fi
  1264│ }
  1265│ 
  1266│ # 主监控循环
  1267│ log "🚀 启动 Git 自动提交监控..."
  1268│ log "监控目录: $PROJECT_PATH"
  1269│ 
  1270│ while true; do
  1271│     # 检查项目目录是否存在
  1272│     if [ ! -d "$PROJECT_PATH/.git" ]; then
  1273│         log "❌ 错误: 不是 Git 仓库"
  1274│         sleep 30
  1275│         continue
  1276│     fi
  1277│     
  1278│     cd "$PROJECT_PATH"
  1279│     
  1280│     # 定期检查（每30秒）
  1281│     check_and_commit
  1282│     
  1283│     # 等待下一次检查
  1284│     sleep 30
  1285│ done
  1286│ EOF
  1287│ 
  1288│ chmod +x git-auto-commit-pro.sh
  1289│ ```
