+def classify(filename):
+    """根据文件名返回分类"""
+    scores = {}
+    for cat, cfg in CATEGORIES.items():
+        score = 0
+        # 排除规则
+        excluded = False
+        for ex in cfg.get("exclude", []):
+            if ex.lower() in filename.lower():
+                excluded = True
+                break
+        if excluded:
+            continue
+        
+        for kw in cfg["keywords"]:
+            if kw.lower() in filename.lower():
+                score += 1
+        scores[cat] = score
+    
+    if not scores:
+        return None
+    
+    best = max(scores, key=scores.get)
+    if scores[best] == 0:
+        return None
+    return best
+
