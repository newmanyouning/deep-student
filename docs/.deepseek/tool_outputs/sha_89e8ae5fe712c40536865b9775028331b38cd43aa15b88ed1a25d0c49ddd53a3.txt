+# 标签映射：基于文件名的关键词自动打标签
+def get_tags(filename):
+    f = filename.lower()
+    tags = []
+    if any(k in f for k in ['错误','报错','bug','error','失败','错误','无法','问题排查','异常']):
+        tags.append('报错排查')
+    if any(k in f for k in ['代码','源码','重构','编程','程序','开发','修复','模块']):
+        tags.append('代码调试')
+    if any(k in f for k in ['提示词','prompt','方案','设计','构思','生成','规划']):
+        tags.append('方案构思')
+    if any(k in f for k in ['安装','使用','下载','配置','命令','运行','操作','工具']):
+        tags.append('工具使用')
+    if not tags:
+        tags.append('知识查询')
+    return ', '.join(tags)
+
+# 从文件名提取主题关键词
+def get_keywords(filename):
+    # 移除扩展名和路径
+    name = os.path.splitext(os.path.basename(filename))[0]
+    # 取前几个有意义的词
+    parts = re.split(r'[与和及的，。、]', name)
+    kw = []
+    for p in parts:
+        p = p.strip()
+        if len(p) >= 2 and len(p) <= 8 and not p.startswith('#'):
+            kw.append(p)
+            if len(kw) >= 3:
+                break
+    return '、'.join(kw[:3]) if kw else name[:15]
+
+# 从文件名生成一句话摘要
+def get_summary(filename, folder):
+    name = os.path.splitext(os.path.basename(filename))[0]
+    name = re.sub(r'[#\d+\.\s]+', '', name)
+    if len(name) > 30:
+        name = name[:30] + '…'
+    # 根据文件夹名添加前缀
+    cat_map = {
+        '01': 'AI/编程', '02': '系统/软件', '03': '健康/医学',
+        '04': '游戏/娱乐', '05': '数学/科学', '06': '苹果设备',
+        '07': '生活/消费', '08': '职业/职场', '09': '硬件/工程',
+        '10': '人文/知识'
+    }
+    prefix = cat_map.get(folder[:2], '')
+    return f'[{prefix}] 关于"{name}"的咨询对话'
