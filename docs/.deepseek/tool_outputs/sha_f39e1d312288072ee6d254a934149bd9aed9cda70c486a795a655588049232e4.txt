  5618│ 1. 针对每个改进点修改提示词

  5651│ #### 5.5.1 基于梯度的提示词优化
  5652│ 
  5653│ ```python
  5654│ # 软提示词优化（Soft Prompt Tuning）
  5655│ import torch
  5656│ import torch.nn as nn
  5657│ 
  5658│ class SoftPromptOptimizer:
  5659│     def __init__(self, llm, num_virtual_tokens=20, embedding_dim=768):
  5660│         """
  5661│         学习连续的提示词嵌入，而非离散文本
  5662│         """
  5663│         self.llm = llm
  5664│         self.num_virtual_tokens = num_virtual_tokens
  5665│         
  5666│         # 可学习的提示词嵌入
  5667│         self.soft_prompts = nn.Parameter(
  5668│             torch.randn(num_virtual_tokens, embedding_dim) * 0.02
  5669│         )
  5670│         self.optimizer = torch.optim.Adam([self.soft_prompts], lr=1e-3)
  5671│     
  5672│     def forward(self, input_ids, labels):
  5673│         # 将软提示词与输入拼接
  5674│         soft_embeds = self.soft_prompts.unsqueeze(0).expand(
  5675│             input_ids.shape[0], -1, -1
  5676│         )
  5677│         input_embeds = self.llm.get_input_embeddings()(input_ids)
  5678│         combined_embeds = torch.cat([soft_embeds, input_embeds], dim=1)
  5679│         
  5680│         # 前向传播
  5681│         outputs = self.llm(inputs_embeds=combined_embeds, labels=labels)
  5682│         loss = outputs.loss
  5683│         
  5684│         # 反向传播
  5685│         loss.backward()
  5686│         self.optimizer.step()
  5687│         
  5688│         return loss.item()
  5689│     
  5690│     def get_prompt_text(self, tokenizer):
  5691│         """将学习到的嵌入转换回文本（近似）"""
  5692│         # 找最近的token
  5693│         distances = torch.cdist(
  5694│             self.soft_prompts, 
  5695│             self.llm.get_input_embeddings().weight
  5696│         )
  5697│         nearest_token_ids = distances.argmin(dim=-1)
  5698│         return tokenizer.decode(nearest_token_ids)
  5699│ ```

  5702│ 
  5703│ ```python
  5704│ # 使用LLM作为优化器
  5705│ class LLMPromptOptimizer:
  5706│     def __init__(self, base_llm, optimizer_llm):
  5707│         self.base_llm = base_llm
  5708│         self.optimizer_llm = optimizer_llm
  5709│         self.history = []
  5710│     
  5711│     def optimize(self, task_description, initial_prompt, test_cases, 
  5712│                  num_iterations=10):
  5713│         current_prompt = initial_prompt
  5714│         
  5715│         for i in range(num_iterations):
  5716│             # 评估当前提示词
  5717│             results = self.evaluate(current_prompt, test_cases)
  5718│             score = results['avg_score']
  5719│             
  5720│             # 记录历史
  5721│             self.history.append({
  5722│                 'iteration': i,
  5723│                 'prompt': current_prompt,
  5724│                 'score': score,
  5725│                 'results': results
  5726│             })
  5727│             
  5728│             # 如果足够好，停止
  5729│             if score > 0.95:
  5730│                 break
  5731│             
  5732│             # 生成优化建议
  5733│             optimization_prompt = f"""
  5734│             任务: {task_description}
  5735│             当前提示词: {current_prompt}
  5736│             测试结果: {results['details']}
  5737│             失败案例: {results['failures']}
  5738│             
  5739│             请分析问题，并提供3个具体的改进建议。
  5740│             每个建议应包含：
  5741│             1. 问题诊断
  5742│             2. 修改后的提示词版本
  5743│             3. 预期的改进效果
  5744│             
  5745│             只输出改进建议，不要有其他内容。
  5746│             """
  5747│             
  5748│             suggestions = self.optimizer_llm.generate(optimization_prompt)
  5749│             
  5750│             # 应用最佳建议
  5751│             best_suggestion = self.select_best_suggestion(suggestions, test_cases)
  5752│             current_prompt = best_suggestion['new_prompt']
  5753│         
  5754│         return current_prompt, self.history
  5755│     
  5756│     def evaluate(self, prompt, test_cases):
  5757│         results = []
  5758│         failures = []
  5759│         
  5760│         for case in test_cases:
  5761│             output = self.base_llm.generate(prompt, input=case['input'])
  5762│             score = self.calculate_score(output, case['expected'])
  5763│             results.append(score)
  5764│             
  5765│             if score < 0.8:  # 失败阈值
  5766│                 failures.append({
  5767│                     'input': case['input'],
  5768│                     'output': output,
  5769│                     'expected': case['expected'],
  5770│                     'score': score
  5771│                 })
  5772│         
  5773│         return {
  5774│             'avg_score': sum(results) / len(results),
  5775│             'details': results,
  5776│             'failures': failures
  5777│         }
  5778│ ```

  5780│ #### 5.5.3 遗传算法优化
  5781│ 
  5782│ ```python
  5783│ # 使用遗传算法进化提示词
  5784│ import random
  5785│ import numpy as np
  5786│ 
  5787│ class GeneticPromptOptimizer:
  5788│     def __init__(self, llm, population_size=20, mutation_rate=0.1):
  5789│         self.llm = llm
  5790│         self.population_size = population_size
  5791│         self.mutation_rate = mutation_rate
  5792│     
  5793│     def initialize_population(self, seed_prompts):
  5794│         """初始化种群"""
  5795│         population = seed_prompts.copy()
  5796│         # 随机生成剩余的
  5797│         while len(population) < self.population_size:
  5798│             population.append(self.random_prompt())
  5799│         return population
  5800│     
  5801│     def fitness(self, prompt, test_cases):
  5802│         """评估适应度"""
  5803│         scores = []
  5804│         for case in test_cases:
  5805│             output = self.llm.generate(prompt, input=case['input'])
  5806│             score = self.evaluate_output(output, case['expected'])
  5807│             scores.append(score)
  5808│         return np.mean(scores)
  5809│     
  5810│     def crossover(self, parent1, parent2):
  5811│         """交叉操作"""
  5812│         # 在随机点分割并组合
  5813│         split_point = random.randint(0, min(len(parent1), len(parent2)))
  5814│         child = parent1[:split_point] + parent2[split_point:]
  5815│         return child
  5816│     
  5817│     def mutate(self, prompt):
  5818│         """变异操作"""
  5819│         if random.random() < self.mutation_rate:
  5820│             # 随机修改：添加/删除/替换
  5821│             operations = ['add', 'delete', 'replace', 'rephrase']
  5822│             op = random.choice(operations)
  5823│             
  5824│             if op == 'rephrase':
  5825│                 # 使用LLM改写
  5826│                 rephrase_prompt = f"请改写以下提示词，保持核心意图但换个表达方式：{prompt}"
  5827│                 return self.llm.generate(rephrase_prompt)
  5828│             elif op == 'add':
  5829│                 # 添加随机指令
  5830│                 additions = ["请确保输出简洁。", "使用专业术语。", "逐步推理。"]
  5831│                 return prompt + " " + random.choice(additions)
  5832│             # ... 其他操作
  5833│         return prompt
  5834│     
  5835│     def evolve(self, test_cases, generations=20):
  5836│         population = self.initialize_population(self.seed_prompts)
  5837│         
  5838│         for gen in range(generations):
  5839│             # 评估适应度
  5840│             fitness_scores = [
  5841│                 self.fitness(prompt, test_cases) 
  5842│                 for prompt in population
  5843│             ]
  5844│             
  5845│             # 选择（锦标赛选择）
  5846│             new_population = []
  5847│             for _ in range(self.population_size):
  5848│                 # 随机选3个，选最好的
  5849│                 tournament = random.sample(
  5850│                     list(zip(population, fitness_scores)), 
  5851│                     k=3
  5852│                 )
  5853│                 winner = max(tournament, key=lambda x: x[1])[0]
  5854│                 new_population.append(winner)
  5855│             
  5856│             # 交叉和变异
  5857│             for i in range(0, self.population_size, 2):
  5858│                 if i+1 < self.population_size:
  5859│                     child1 = self.crossover(new_population[i], new_population[i+1])
  5860│                     child2 = self.crossover(new_population[i+1], new_population[i])
  5861│                     new_population[i] = self.mutate(child1)
  5862│                     new_population[i+1] = self.mutate(child2)
  5863│             
  5864│             population = new_population
  5865│             
  5866│             # 记录最佳
  5867│             best_idx = np.argmax(fitness_scores)
  5868│             best_prompt = population[best_idx]
  5869│             best_fitness = fitness_scores[best_idx]
  5870│             
  5871│             print(f"Generation {gen}: Best fitness = {best_fitness:.3f}")
  5872│         
  5873│         return best_prompt
  5874│ ```
