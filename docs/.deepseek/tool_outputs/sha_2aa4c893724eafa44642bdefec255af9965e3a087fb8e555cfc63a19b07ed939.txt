import json, os, re, sys
sys.stdout.reconfigure(encoding='utf-8')

base = r'C:\Users\1\Downloads\deepseek_data-2026-05-17\output_md'

with open(os.path.join(base, '_scan_result.json'), 'r', encoding='utf-8') as f:
    data = json.load(f)

# Files with >= 1500 lines (indices 0-40)
big_files = [e for e in data['all'] if e['lines'] >= 1500]

results = []
for idx, entry in enumerate(big_files):
    fp = os.path.join(base, entry['path'])
    try:
        lines = []
        with open(fp, 'r', encoding='utf-8', errors='ignore') as fh:
            for i, line in enumerate(fh):
                if i < 40:
                    lines.append(line.strip())
                elif i >= 40:
                    break
        head = '\n'.join(l for l in lines if l and not l.startswith('#'))
        
        # Get tail
        tail_lines = []
        with open(fp, 'r', encoding='utf-8', errors='ignore') as fh:
            all_l = fh.readlines()
            tail_start = max(0, len(all_l) - 40)
            for l in all_l[tail_start:]:
                tail_lines.append(l.strip())
        tail = '\n'.join(l for l in tail_lines if l and not l.startswith('#') and not l.startswith('---'))
        
        results.append({
            'idx': idx,
            'path': entry['path'],
            'lines': entry['lines'],
            'head': head[:300],
            'tail': tail[:300],
        })
    except Exception as e:
        results.append({
            'idx': idx,
            'path': entry['path'],
            'lines': entry['lines'],
            'head': f'ERROR: {e}',
            'tail': '',
        })

with open(os.path.join(base, '_bigfiles_snippets.json'), 'w', encoding='utf-8') as f:
    json.dump(results, f, ensure_ascii=False, indent=2)

print(f'Extracted {len(results)} files')
for r in results:
    print(f"\n[{r['idx']}] {r['lines']}L — {os.path.basename(r['path'])}")
    print(f"  HEAD: {r['head'][:120]}...")
    print(f"  TAIL: {r['tail'][:120]}...")
