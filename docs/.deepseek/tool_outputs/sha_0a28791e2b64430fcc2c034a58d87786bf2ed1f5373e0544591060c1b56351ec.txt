   867│  不过，有一个Git钩子叫做`post-commit`，它是在提交后运行的。但我们需要的是在文件修改后，而不是提交后。因此，我们可以考虑使用操作系统的文件监控功能。
   868│ 
   869│  但如果我们坚持使用Git钩子，并且希望是在每次提交后自动推送（这符合“同步”的部分含义，但不符合“修改后立即”），那么我们可以使用`post-commit`钩子。

   899│  这里，我将给出一个使用Node.js和chokidar的示例。
   900│ 
   901│  步骤：
   902│ 
   903│  1. 初始化一个Node.js项目（在Git仓库根目录）：
   904│     ```
   905│     npm init -y
   906│     ```
   907│  2. 安装chokidar：
   908│     ```
   909│     npm install chokidar
   910│     ```
   911│  3. 创建一个监控脚本`auto-commit.js`：
   912│     ```javascript
   913│     const chokidar = require('chokidar');
   914│     const { exec } = require('child_process');
   915│     const path = require('path');
   916│ 
   917│     // 获取当前目录作为监控目录
   918│     const watchDir = process.cwd();
   919│ 
   920│     // 初始化监控
   921│     const watcher = chokidar.watch(watchDir, {
   922│       ignored: /(^|[\/\\])\../, // 忽略隐藏文件
   923│       persistent: true,
   924│       ignoreInitial: true, // 忽略初始扫描时的文件事件
   925│       awaitWriteFinish: {
   926│         stabilityThreshold: 500,
   927│         pollInterval: 100
   928│       }
   929│     });
   930│ 
   931│     // 防抖函数，避免短时间内多次提交
   932│     let timeout;
   933│     function debounceCommit() {
   934│       clearTimeout(timeout);
   935│       timeout = setTimeout(() => {
   936│         console.log('检测到文件变化，正在自动提交...');
   937│         exec('git add .', (err, stdout, stderr) => {
   938│           if (err) {
   939│             console.error('添加文件出错:', err);
   940│             return;
   941│           }
   942│           exec('git commit -m "自动提交"', (err, stdout, stderr) => {
   943│             if (err) {
   944│               console.error('提交出错:', err);
   945│               return;
   946│             }
   947│             console.log('自动提交完成:', stdout);
   948│           });
   949│         });
   950│       }, 2000); // 2秒后执行，避免频繁提交
   951│     }
   952│ 
   953│     // 监听变化
   954│     watcher.on('all', (event, path) => {
   955│       console.log(`文件 ${path} 发生了 ${event} 事件`);
   956│       debounceCommit();
   957│     });
   958│ 
   959│     console.log(`开始监控 ${watchDir} 目录...`);
   960│     ```
   961│  4. 运行这个脚本：
   962│     ```
   963│     node auto-commit.js
   964│     ```

   984│  这里是一个简单的PowerShell脚本示例（Windows）：
   985│ 
   986│  ```powershell
   987│  $watcher = New-Object System.IO.FileSystemWatcher
   988│  $watcher.Path = "C:\path\to\your\repo"
   989│  $watcher.IncludeSubdirectories = $true
   990│  $watcher.EnableRaisingEvents = $true  
   991│ 
   992│  $action = {
   993│      $path = $Event.SourceEventArgs.FullPath
   994│      $changeType = $Event.SourceEventArgs.ChangeType
   995│      $logline = "$(Get-Date), $changeType, $path"
   996│      Add-Content "C:\log.txt" -Value $logline
   997│ 
   998│      # 防抖，等待2秒无新事件再提交
   999│      Start-Sleep -Seconds 2
  1000│      git add .
  1001│      git commit -m "自动提交"
  1002│  }
  1003│ 
  1004│  Register-ObjectEvent $watcher "Created" -Action $action
  1005│  Register-ObjectEvent $watcher "Changed" -Action $action
  1006│  Register-ObjectEvent $watcher "Deleted" -Action $action
  1007│ 
  1008│  while ($true) {Start-Sleep 1}
  1009│  ```
  1010│ 
  1011│  将上述PowerShell脚本保存为`watch.ps1`，然后以管理员权限运行。
  1012│

  1034│ ## Git 钩子实现自动同步的完整方案
  1035│ 
  1036│ ### 一、Git 钩子的创建位置和机制
  1037│ 
  1038│ **钩子存放位置：** `.git/hooks/` 目录下
  1039│ 
  1040│ ```bash
  1041│ # 查看所有可用钩子
  1042│ ls .git/hooks/
  1043│ 
  1044│ # 常见钩子文件：
  1045│ # pre-commit      # 提交前运行
  1046│ # post-commit     # 提交后运行
  1047│ # pre-push        # 推送前运行
  1048│ # post-checkout   # 检出后运行
  1049│ ```
  1050│ 
  1051│ ### 二、创建自动提交钩子
  1052│ 
  1053│ #### 方案1：最简单的 post-commit 钩子（手动触发后自动提交）
  1054│ ```bash
  1055│ # 1. 进入钩子目录
  1056│ cd .git/hooks
  1057│ 
  1058│ # 2. 创建 post-commit 钩子
  1059│ cat > post-commit << 'EOF'
  1060│ #!/bin/bash
  1061│ echo "检测到新提交，自动保存所有修改..."
  1062│ git add .
  1063│ git commit --amend -C HEAD --no-edit 2>/dev/null || true
  1064│ EOF
  1065│ 
  1066│ # 3. 给予执行权限
  1067│ chmod +x post-commit
  1068│ ```
  1069│ 
  1070│ #### 方案2：创建文件监控钩子（推荐）
  1071│ 
  1072│ 创建一个独立的监控脚本：
  1073│ 
  1074│ ```bash
  1075│ # 1. 在项目根目录创建监控脚本
  1076│ cat > git-auto-commit.sh << 'EOF'
  1077│ #!/bin/bash
  1078│ # 自动提交监控脚本
  1079│ echo "🔍 开始监控文件变化..."
  1080│ 
  1081│ # 设置监控路径
  1082│ WATCH_PATH="."
  1083│ IGNORE_PATTERN='\.git/|node_modules/|\.log$|\.tmp$'
  1084│ 
  1085│ # 持续监控文件变化
  1086│ while true; do
  1087│     # 等待文件变化（macOS/Linux）
  1088│     if command -v fswatch &> /dev/null; then
  1089│         # macOS 使用 fswatch
  1090│         fswatch -1 -r -e "$IGNORE_PATTERN" "$WATCH_PATH" > /dev/null
  1091│     elif command -v inotifywait &> /dev/null; then
  1092│         # Linux 使用 inotifywait
  1093│         inotifywait -r -e modify,create,delete --exclude "$IGNORE_PATTERN" "$WATCH_PATH" > /dev/null
  1094│     else
  1095│         # 简单轮询（兼容所有系统）
  1096│         find "$WATCH_PATH" -type f -newer /tmp/last_check 2>/dev/null | head -1
  1097│         touch /tmp/last_check
  1098│         sleep 2
  1099│         continue
  1100│     fi
  1101│     
  1102│     # 等待一小段时间，避免多次触发
  1103│     sleep 1
  1104│     
  1105│     # 检查是否有未提交的修改
  1106│     if ! git diff --quiet || ! git diff --cached --quiet; then
  1107│         echo "📝 检测到文件变化，正在自动提交..."
  1108│         
  1109│         # 添加所有修改
  1110│         git add . 2>/dev/null
  1111│         
  1112│         # 生成时间戳
  1113│         TIMESTAMP=$(date "+%Y-%m-%d %H:%M:%S")
  1114│         
  1115│         # 提交
  1116│         git commit -m "🔄 自动提交: $TIMESTAMP" --quiet
  1117│         
  1118│         echo "✅ 已提交: $TIMESTAMP"
  1119│     fi
  1120│ done
  1121│ EOF
  1122│ 
  1123│ # 2. 给予执行权限
  1124│ chmod +x git-auto-commit.sh
  1125│ ```
