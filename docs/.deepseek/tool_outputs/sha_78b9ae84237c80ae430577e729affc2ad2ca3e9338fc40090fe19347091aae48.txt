+#!/usr/bin/env python3
+"""
+知识库构建进度更新脚本
+用途：读取 project_progress.md 当前进度，更新统计数字
+用法：
+  python update_progress.py status          # 仅读取当前进度
+  python update_progress.py update <已完成数> <总文件数>  # 更新进度
+"""
+
+import re
+import sys
+import os
+from pathlib import Path
+
+PROGRESS_FILE = Path(__file__).parent / "project_progress.md"
+
+
+def read_status():
+    """读取 project_progress.md 并提取关键进度信息"""
+    if not PROGRESS_FILE.exists():
+        print(f"[错误] 未找到 {PROGRESS_FILE}")
+        return None
+
+    content = PROGRESS_FILE.read_text(encoding="utf-8")
+
+    # 提取总体统计
+    total_match = re.search(r"- 总文件数:\s*(\d+)", content)
+    done_match = re.search(r"- 已完成:\s*(\d+)", content)
+    remain_match = re.search(r"- 待处理:\s*(\d+)", content)
+    coverage_match = re.search(r"- 覆盖率:\s*(\d+)%", content)
+
+    total = int(total_match.group(1)) if total_match else 0
+    done = int(done_match.group(1)) if done_match else 0
+    remain = int(remain_match.group(1)) if remain_match else 0
+    coverage = int(coverage_match.group(1)) if coverage_match else 0
+
+    # 提取各领域完成情况
+    domains = {}
+    domain_pattern = re.findall(r"\|\s*([^|]+)\s*\|\s*(\d+)/(\d+)\s*\|", content)
+    for name, done_n, total_n in domain_pattern:
+        name = name.strip()
+        domains[name] = {"done": int(done_n), "total": int(total_n)}
+
+    # 提取最后一批处理记录
+    batch_match = re.search(r"## 本轮处理记录.*?\n- 处理数量:\s*(\d+)", content, re.DOTALL)
+    batch_info = ""
+    if batch_match:
+        lines_after = content[content.find("## 本轮处理记录"):]
+        info_lines = lines_after.split("\n")[1:6]
+        batch_info = "\n".join(l.strip() for l in info_lines if l.strip())
+
+    # 提取下一步待处理文件
+    next_batch = []
+    in_next = False
+    for line in content.split("\n"):
+        if "下一批待处理的中文件" in line:
+            in_next = True
+            continue
+        if in_next and line.strip().startswith("```"):
+            break
+        if in_next and line.strip().startswith(("1.", "2.", "3.", "4.", "5.")):
+            next_batch.append(line.strip())
+
+    return {
+        "total": total,
+        "done": done,
+        "remain": remain,
+        "coverage": coverage,
+        "domains": domains,
+        "batch_info": batch_info,
+        "next_batch": next_batch,
+    }
+
+
+def update_status(done_new: int, total_new: int):
+    """更新 project_progress.md 中的进度数字"""
+    content = PROGRESS_FILE.read_text(encoding="utf-8")
+
+    coverage_new = round(done_new / total_new * 100) if total_new > 0 else 0
+    remain_new = total_new - done_new
+
+    # 替换统计行
+    content = re.sub(r"- 总文件数:\s*\d+", f"- 总文件数: {total_new}", content)
+    content = re.sub(r"- 已完成:\s*\d+", f"- 已完成: {done_new}", content)
+    content = re.sub(r"- 待处理:\s*\d+", f"- 待处理: {remain_new}", content)
+    content = re.sub(r"- 覆盖率:\s*\d+%", f"- 覆盖率: {coverage_new}%", content)
+
+    PROGRESS_FILE.write_text(content, encoding="utf-8")
+    print(f"[成功] 已更新: 总{total_new}, 已完成{done_new}, 待处理{remain_new}, 覆盖率{coverage_new}%")
+    return True
+
+
+def main():
+    if len(sys.argv) < 2:
+        print("用法:")
+        print("  python update_progress.py status              # 读取当前进度")
+        print("  python update_progress.py update <完成> <总数> # 更新进度")
+        return
+
+    command = sys.argv[1]
+
+    if command == "status":
+        status = read_status()
+        if not status:
+            return
+        print("=" * 60)
+        print("📊 知识库构建进度")
+        print("=" * 60)
+        print(f"总文件数: {status['total']}")
+        print(f"已完成:   {status['done']}")
+        print(f"待处理:   {status['remain']}")
+        print(f"覆盖率:   {status['coverage']}%")
+        print("-" * 60)
+        if status["domains"]:
+            print("领域完成情况:")
+            for name, info in sorted(status["domains"].items()):
+                bar = "█" * info["done"] + "░" * (info["total"] - info["done"])
+                print(f"  {name:12s} {info['done']:3d}/{info['total']:<3d} {bar}")
+        if status["next_batch"]:
+            print("-" * 60)
+            print("下一批待处理:")
+            for item in status["next_batch"]:
+                print(f"  {item}")
+        print("=" * 60)
+
+    elif command == "update":
+        if len(sys.argv) < 4:
+            print("[错误] 用法: update_progress.py update <已完成数> <总文件数>")
+            return
+        done_new = int(sys.argv[2])
+        total_new = int(sys.argv[3])
+        update_status(done_new, total_new)
+
+    else:
+        print(f"[错误] 未知命令: {command}")
+
+
+if __name__ == "__main__":
+    main()
